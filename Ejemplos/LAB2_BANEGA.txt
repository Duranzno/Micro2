void ingresar()
{ short cont=0,Tecla=0,Especial=0,Pulsada_soltada=0;
     while(cont<4)
     {        while(Ps2_Key_Read(&Tecla,&Especial,&Pulsada_soltada)==0);
             if(Pulsada_soltada==1 && Especial==0)
             {        switch(cont)
                      {case 0:
                      Glcd_Write_Char(Tecla,0,3,1);
                      IC2CON1bits.ICM=4;
                      IC1CON1bits.ICM=4;
                      cont++;
                      pulso();
                      break;

                      case 1:
                      Glcd_Write_Char(Tecla,6,3,1);

                      pulso();
                      break;

                      pulso();
                      break;

                      case 3:
                      Glcd_Write_Char(Tecla,18,3,1);
                      cont++;
                      pulso();
                      break;
                      }
              }

      }
}
